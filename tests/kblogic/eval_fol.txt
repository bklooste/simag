	( scum[$West,u=1] && good[$West,u=0] )

BLOCK
	( drugDealer[$West,u=1] => ( scum[$West,u=1] && good[$West,u=0] ) )
/BLOCK
# Test 2.1/02 -> True (true => true)
BLOCK
	( drugDealer[$West,u=1] => ( scum[$West,u=1] && good[$West,u=0] ) )
	( drugDealer[$West,u=1] )
	( scum[$West,u=1] )
	( good[$West,u=0] )
/BLOCK
# Test 2.2/03 -> True (false => true)
BLOCK
	( drugDealer[$West,u=1] => ( scum[$West,u=1] && good[$West,u=0] ) )
	( drugDealer[$West,u=0] )
	( scum[$West,u=1] )
	( good[$West,u=0] )
/BLOCK
# Test 2.3/04 -> false (true => false)
BLOCK
	( drugDealer[$West,u=1] => ( scum[$West,u=1] && good[$West,u=0] ) )
	( drugDealer[$West,u=1] )
	( scum[$West,u=0] )
	( good[$West,u=1] )
/BLOCK
# Test 2.4/05 -> true (false => false)
BLOCK
	( drugDealer[$West,u=1] => ( scum[$West,u=1] && good[$West,u=0] ) )
	( drugDealer[$West,u=0] )
	( scum[$West,u=0] )
	( good[$West,u=1] )
/BLOCK
# Test 3 => Evaluate an equivalence (IFF) statements
# Test 3.0/06 ->  Cannot be solved: None (none <=> true)
BLOCK
	( drugDealer[$West,u=1] <=> ( scum[$West,u=1] && good[$West,u=0] ) )
	( scum[$West,u=1] )
	( good[$West,u=0] )
/BLOCK
# Test 3.1/07 -> Is false (false <=> true )
BLOCK
	( drugDealer[$West,u=1] <=> ( scum[$West,u=1] && good[$West,u=0] ) )

	( drugDealer[$West,u=0] )
/BLOCK

BLOCK
	( ( drugDealer[$West,u=1] <=> scum[$West,u=1] ) && good[$West,u=0] )
	( scum[$West,u=1] )

	( drugDealer[$West,u=1] <=> ( scum[$West,u=1] && good[$West,u=0] ) )
	( scum[$West,u=0] )
	( good[$West,u=1] )
	( drugDealer[$West,u=0] )
/BLOCK
